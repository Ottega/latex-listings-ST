	values : ARRAY[1..1000] OF REAL;

	index : INT;
END_VAR;

\begin{lstlisting}[language=ST, caption=function block]

FOR Index := 1 TO 1000 BY 1 DO
	sum := sum + values[index];
END_FOR;
averageOf1000 := sum / 1000.0;
